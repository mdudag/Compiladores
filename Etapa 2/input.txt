\ EXEMPLO DE CODIGO PREX
   Objetivo: receber dois numeros, realizar a soma e exibir no video.
   Entrada: dois numeros.
  Saida: o resultado da soma destes dois numeros.
 \
PROGRAMA PREX {
	FLOAT n1, n2, soma;
	ESCREVER textoEntrada;
	LER n1;
	ESCREVER textoEntrada;
	LER n2;
	soma = + n1 n2;
	ESCREVER textoSaida;
	ESCREVER soma;
}
