[\ EXEMPLO DE CODIGO PREX
   Objetivo: receber dois numeros, realizar a soma e exibir no video.
   Entrada: dois numeros.
  Saida: o resultado da soma destes dois numeros.
 \] -> COMENTARIO

[PROGRAMA] -> PALAVRA RESERVADA
[PREX] -> PALAVRA RESERVADA
[{] -> CHAVE A ESQUERDA

[FLOAT n1, n2, soma;] -> VARIAVEL TIPO FLOAT

[LISTA] -> PALAVRA RESERVADA
[CHAR] -> PALAVRA RESERVADA
[textoEntra] -> IDENTIFICADOR
[da] -> IDENTIFICADOR
[(] -> PARENTESE A ESQUERDA
[2] -> ERRO
[5] -> ERRO
[)] -> PARENTESE A DIREITA
[,] -> ERRO
[textoSaida] -> IDENTIFICADOR
[(] -> PARENTESE A ESQUERDA
[1] -> TIPO BOOL
[0] -> TIPO BOOL
[)] -> PARENTESE A DIREITA
[;] -> SINAL DE FIM DE INSTRUCAO

[textoEntra] -> IDENTIFICADOR
[da] -> IDENTIFICADOR
[=] -> OPERADOR DE ATRIBUICAO
["Digite um numero real: \n"] -> TIPO LISTA CHAR
[;] -> SINAL DE FIM DE INSTRUCAO

[textoSaida] -> IDENTIFICADOR
[=] -> OPERADOR DE ATRIBUICAO
["A soma e: "] -> TIPO LISTA CHAR
[;] -> SINAL DE FIM DE INSTRUCAO

[ESCREVER] -> PALAVRA RESERVADA
[textoEntra] -> IDENTIFICADOR
[da] -> IDENTIFICADOR
[;] -> SINAL DE FIM DE INSTRUCAO

[LER] -> PALAVRA RESERVADA
[n1] -> IDENTIFICADOR
[;] -> SINAL DE FIM DE INSTRUCAO

[ESCREVER] -> PALAVRA RESERVADA
[textoEntra] -> IDENTIFICADOR
[da] -> IDENTIFICADOR
[;] -> SINAL DE FIM DE INSTRUCAO

[LER] -> PALAVRA RESERVADA
[n2] -> IDENTIFICADOR
[;] -> SINAL DE FIM DE INSTRUCAO

[soma] -> IDENTIFICADOR
[=] -> OPERADOR DE ATRIBUICAO
[+] -> OPERADOR DE SOMA
[n1] -> IDENTIFICADOR
[n2] -> IDENTIFICADOR
[;] -> SINAL DE FIM DE INSTRUCAO

[ESCREVER] -> PALAVRA RESERVADA
[textoSaida] -> IDENTIFICADOR
[;] -> SINAL DE FIM DE INSTRUCAO

[ESCREVER] -> PALAVRA RESERVADA
[soma] -> IDENTIFICADOR
[;] -> SINAL DE FIM DE INSTRUCAO

[}] -> CHAVE A DIREITA
