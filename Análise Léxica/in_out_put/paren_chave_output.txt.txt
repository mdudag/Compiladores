[BOOL] -> PALAVRA RESERVADA
[n] -> IDENTIFICADOR
[=] -> OPERADOR DE ATRIBUICAO
[1] -> TIPO BOOL
[;] -> SINAL DE FIM DE INSTRUCAO

[FLOAT] -> PALAVRA RESERVADA
[B2] -> IDENTIFICADOR
[;] -> SINAL DE FIM DE INSTRUCAO

[SE] -> PALAVRA RESERVADA
[(] -> PARENTESE A ESQUERDA
[n] -> IDENTIFICADOR
[)] -> PARENTESE A DIREITA
[ENTAO] -> PALAVRA RESERVADA
[{] -> CHAVE A ESQUERDA

[B2] -> IDENTIFICADOR
[=] -> OPERADOR DE ATRIBUICAO
[-45.67] -> TIPO FLOAT
[;] -> SINAL DE FIM DE INSTRUCAO

[CHAR] -> PALAVRA RESERVADA
[c] -> IDENTIFICADOR
[=] -> OPERADOR DE ATRIBUICAO
['@'] -> TIPO CHAR
[;] -> SINAL DE FIM DE INSTRUCAO

[}] -> CHAVE A DIREITA

[SENAO] -> PALAVRA RESERVADA
[{] -> CHAVE A ESQUERDA

[B2] -> IDENTIFICADOR
[=] -> OPERADOR DE ATRIBUICAO
[0.0] -> TIPO FLOAT
[;] -> SINAL DE FIM DE INSTRUCAO

[}] -> CHAVE A DIREITA
